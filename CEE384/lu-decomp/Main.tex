	\vec{L} \vec{U} = \vec{A}
\end{equation}


\begin{align}
	\vec{A} \vec{x} &= \vec{b} \\
	\vec{P} \vec{A} \vec{x} &= \vec{P} \vec{b} \equiv \vec{d}\\
	\vec{L} \vec{y} &= \vec{d}, \vec{y} \equiv \vec{U} \vec{x}\\
	\vec{U} \vec{x} &= \vec{y}
\end{align}
